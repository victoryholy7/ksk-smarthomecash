        window.exportData = function () {
            const json = JSON.stringify(appData, null, 2);
            const blob = new Blob([json], { type: 'application/json' });
            const url  = URL.createObjectURL(blob);
            const a    = document.createElement('a');

        window.importData = function (event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function (e) {
                try {
                    const imported = JSON.parse(e.target.result);
                    if (!imported.transactions || !imported.categories) {
                        Swal.fire('ข้อผิดพลาด', 'ไฟล์ JSON ไม่ถูกต้อง', 'error');
                        return;
                    }
                    Swal.fire({
                        title: 'ยืนยันการนำเข้าข้อมูล?',
                        text: 'ข้อมูลปัจจุบันจะถูกแทนที่ด้วยข้อมูลจากไฟล์',
                        icon: 'warning',
                        showCancelButton: true,
                        confirmButtonText: 'นำเข้าเลย',
                        cancelButtonText: 'ยกเลิก'
                    }).then(result => {
                        if (result.isConfirmed) {
                            appData = imported;
                            saveData();
                            initializeUI();
                            // reset file input
                            document.getElementById('importFile').value = '';
                            Swal.fire('สำเร็จ', 'นำเข้าข้อมูลเรียบร้อยแล้ว', 'success');
                        }
                    });
                } catch (err) {
                    Swal.fire('ข้อผิดพลาด', 'ไม่สามารถอ่านไฟล์ได้: ' + err.message, 'error');
                }
            };
            reader.readAsText(file);
        };

        function populateDashboardMonths() {
            const select = $('#dashboardMonth');
            select.empty();

            const months = [...new Set(appData.transactions.map(t => t.date.slice(0, 7)))].sort().reverse();
            const currentMonth = new Date().toISOString().slice(0, 7);
            if (!months.includes(currentMonth)) months.unshift(currentMonth);

            months.forEach(m => {
                const [y, mon] = m.split('-');
                const monthName = new Date(y, mon - 1).toLocaleDateString('th-TH', { month: 'long' });
                select.append(`<option value="${m}">${monthName} ${parseInt(y) + 543}</option>`);
            });
            select.val(currentMonth);
        }

        function updateDashboard() {
            const selectedMonth = $('#dashboardMonth').val();
            const txs     = appData.transactions.filter(t => t.date.startsWith(selectedMonth));
            const income  = txs.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0);
            const expense = txs.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0);
            const balance = income - expense;

            $('#totalIncome').text('฿' + income.toLocaleString());
            $('#totalExpense').text('฿' + expense.toLocaleString());
            $('#balance').text('฿' + balance.toLocaleString());

            updateCharts(txs, selectedMonth);
            updateBudgetProgressCompact(selectedMonth);
            updateYearChart();
        }

        function updateCharts(txs, month) {
            const makeDonut = (ctx, labels, data, colors) => new Chart(ctx, {
                type: 'doughnut',
                data: {
                    labels,
                    datasets: [{ data, backgroundColor: colors, borderWidth: 0, hoverOffset: 10 }]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    cutout: '70%',
                    animation: { duration: 500 },
                    plugins: { legend: { position: 'bottom', labels: { boxWidth: 10, font: { family: 'Prompt' } } } }
                }
            });

            const incCats = {}, expCats = {};
            txs.filter(t => t.type === 'income').forEach(t  => incCats[t.category] = (incCats[t.category] || 0) + t.amount);
            txs.filter(t => t.type === 'expense').forEach(t => expCats[t.category] = (expCats[t.category] || 0) + t.amount);

            const getColor = name => appData.categories.find(c => c.name === name)?.color || '#ccc';

            if (charts.income)  charts.income.destroy();
            if (charts.expense) charts.expense.destroy();

            charts.income  = makeDonut(document.getElementById('incomeChart'),
                Object.keys(incCats), Object.values(incCats), Object.keys(incCats).map(getColor));
            charts.expense = makeDonut(document.getElementById('expenseChart'),
                Object.keys(expCats), Object.values(expCats), Object.keys(expCats).map(getColor));
        }

        function updateBudgetProgressCompact(month) {
            const container = $('#budgetProgressCompact');
            container.empty();

            const budgets = appData.budgets[month] || [];
            if (!budgets.length) {
                container.html('<div class="text-center text-muted my-5"><i class="fas fa-search mb-2"></i><br>ยังไม่มีการตั้งงบประมาณเดือนนี้</div>');
                return;
            }

            budgets.forEach(b => {
                const cat = appData.categories.find(c => c.id == b.categoryId);
                if (!cat) return;

                const actual = appData.transactions
                    .filter(t => t.date.startsWith(month) && t.category === cat.name && t.type === 'expense')
                    .reduce((s, t) => s + t.amount, 0);

                const pct   = (actual / b.amount) * 100;
                const color = pct > 100 ? 'bg-danger' : pct > 80 ? 'bg-warning' : 'bg-success';

                container.append(`
                    <div class="mb-3">
                        <div class="d-flex justify-content-between small mb-1">
                            <span class="fw-bold">${cat.name}</span>
                            <span class="${pct > 100 ? 'text-danger' : 'text-muted'}">฿${actual.toLocaleString()} / ฿${b.amount.toLocaleString()}</span>
                        </div>
                        <div class="progress" style="height: 8px; border-radius: 4px;">
                            <div class="progress-bar ${color}" role="progressbar" style="width: ${Math.min(pct, 100)}%"></div>
                        </div>
                    </div>
                `);
            });
        }

        function renderCalendar() {
            const year  = calendarDate.getFullYear();
            const month = calendarDate.getMonth();

            $('#calendarMonthDisplay').text(`${new Date(year, month).toLocaleDateString('th-TH', { month: 'long' })} ${year + 543}`);

            const grid       = $('#calendarGrid');
            grid.empty();

            const firstDay    = new Date(year, month, 1).getDay();
            const daysInMonth = new Date(year, month + 1, 0).getDate();
            const todayStr    = new Date().toISOString().slice(0, 10);

            for (let i = 0; i < firstDay; i++) grid.append('<div class="calendar-day empty"></div>');

            for (let d = 1; d <= daysInMonth; d++) {
                const dateStr  = `${year}-${String(month + 1).padStart(2, '0')}-${String(d).padStart(2, '0')}`;
                const dailyTxs = appData.transactions.filter(t => t.date === dateStr);
                const dayInc   = dailyTxs.filter(t => t.type === 'income').reduce((s, t) => s + t.amount, 0);
                const dayExp   = dailyTxs.filter(t => t.type === 'expense').reduce((s, t) => s + t.amount, 0);
                const dayBal   = dayInc - dayExp;

                let summaryHtml = '';
                if (dayInc > 0 || dayExp > 0) {
                    summaryHtml = `
                        <div class="day-summary">
                            ${dayInc > 0 ? `<div class="sum-item sum-inc"><span>รับ</span><span>${formatShort(dayInc)}</span></div>` : ''}
                            ${dayExp > 0 ? `<div class="sum-item sum-exp"><span>จ่าย</span><span>${formatShort(dayExp)}</span></div>` : ''}
                            <div class="sum-item sum-bal text-center d-block mt-1">${dayBal >= 0 ? '+' : ''}${formatShort(dayBal)}</div>
                        </div>`;
                }

                grid.append(`
                    <div class="calendar-day ${dateStr === todayStr ? 'today' : ''}" onclick="viewDailyDetail('${dateStr}')">
                        <div class="d-flex justify-content-between">
                            <span class="day-number">${d}</span>
                        </div>
                        ${summaryHtml}
                    </div>`);
            }
        }

        function formatShort(num) {
            return Math.abs(num) >= 1000 ? (num / 1000).toFixed(1) + 'k' : num.toLocaleString();
        }

            tbody.empty();

            const fCat   = $('#filterCategory').val();
            const fStart = $('#filterStartDate').val();
            const fEnd   = $('#filterEndDate').val();

            const data = appData.transactions.slice().reverse().filter(t => {
                if (fCat   && t.category !== fCat)  return false;
                if (fStart && t.date < fStart)       return false;
                if (fEnd   && t.date > fEnd)         return false;
                return true;
            });

            data.forEach(t => {
                const color = t.type === 'income' ? 'success' : 'danger';
                const sign  = t.type === 'income' ? '+' : '-';
                tbody.append(`
                    <tr>
                        <td>${new Date(t.date).toLocaleDateString('th-TH')}</td>
                        <td><span class="badge bg-${color} bg-opacity-10 text-${color}">${t.category}</span></td>
                        <td>${t.description}</td>
                        <td class="text-end text-${color} fw-bold">${sign}฿${t.amount.toLocaleString()}</td>
                        <td class="text-center">
                            <button class="btn btn-sm btn-light text-danger" onclick="deleteTransaction(${t.id})">
                                <i class="fas fa-trash"></i>
                            </button>
                        </td>
                    </tr>`);
            });
        }

        window.deleteTransaction = function (id) {
            Swal.fire({ title: 'ลบรายการนี้?', icon: 'warning', showCancelButton: true })
                .then(r => {
                    if (r.isConfirmed) {
                        appData.transactions = appData.transactions.filter(t => t.id !== id);
                        saveData();
                        loadTransactionHistory();
                        updateDashboard();
                        renderCalendar();
                    }
                });
        };

        window.clearAllData = function () {
            Swal.fire({
                title: 'ยืนยันการล้างข้อมูล?',
                text:  'ข้อมูลทั้งหมดจะหายไป! แนะนำ Export ก่อน',
                icon:  'warning',
                showCancelButton: true,
                confirmButtonColor: '#d33',
                confirmButtonText: 'ล้างข้อมูล',
                cancelButtonText:  'ยกเลิก'
            }).then(r => {
                if (r.isConfirmed) {
                    localStorage.removeItem('financeApp_Pro');
                    location.reload();
                }
            });
        };

        // ─── PROFILE ─────────────────────────────────────────────────────
        window.updateProfileUI = function () {
            const p = appData.profile;
            $('#displayProfileName').text(p.name);
            $('#displayProfileImg').attr('src', p.image || `https://ui-avatars.com/api/?name=${encodeURIComponent(p.name)}&background=4361ee&color=fff`);
        };

        window.editProfile = function () {
            Swal.fire({
                title: 'แก้ไขโปรไฟล์',
                html: `
                    <div class="mb-3">
                        <label class="form-label text-start d-block">ชื่อเรียก (Nickname)</label>
                        <input id="swal-input1" class="form-control" value="${appData.profile.name}">
                    </div>
                    <div class="mb-3">
                        <label class="form-label text-start d-block">อัปโหลดรูปโปรไฟล์</label>
                        <input type="file" id="swal-input-file" class="form-control" accept="image/*">
                    </div>
                    <div class="mb-3">
                        <label class="form-label text-start d-block">หรือ ลิงก์รูปภาพ (URL)</label>
                        <input id="swal-input2" class="form-control" placeholder="https://..." value="${appData.profile.image && !appData.profile.image.startsWith('data:') ? appData.profile.image : ''}">
                        <small class="text-muted d-block text-start mt-1">*เว้นว่างเพื่อใช้รูปอัตโนมัติ</small>
                    </div>`,
                focusConfirm: false,
                showCancelButton: true,
                confirmButtonText: 'บันทึก',
                preConfirm: () => [
                    document.getElementById('swal-input1').value,
                    document.getElementById('swal-input2').value,
                    document.getElementById('swal-input-file').files[0]
                ]
            }).then(result => {
                if (!result.isConfirmed) return;
                const [name, urlInput, fileInput] = result.value;

                const saveProfile = (name, image) => {
                    appData.profile = { name, image };
                    saveData();
                    updateProfileUI();
                    Swal.fire('บันทึกสำเร็จ', '', 'success');
                };

                if (fileInput) {
                    const reader = new FileReader();
                    reader.onload = e => saveProfile(name, e.target.result);
                    reader.readAsDataURL(fileInput);
                } else {
                    saveProfile(name, urlInput);
                }
            });
        };

        // ─── BUDGET ──────────────────────────────────────────────────────
        window.loadBudgetForm = function () {
            if ($('#budgetYearSelect option').length === 0) {
                const y = new Date().getFullYear();
                for (let i = y - 2; i <= y + 2; i++)
                    $('#budgetYearSelect').append(`<option value="${i}" ${i === y ? 'selected' : ''}>${i + 543}</option>`);
                for (let i = 1; i <= 12; i++)
                    $('#budgetMonthSelect').append(`<option value="${String(i).padStart(2,'0')}" ${i === (new Date().getMonth()+1) ? 'selected' : ''}>${new Date(2000,i-1).toLocaleDateString('th-TH',{month:'long'})}</option>`);
            }

            const key   = `${$('#budgetYearSelect').val()}-${$('#budgetMonthSelect').val()}`;
            const tbody = $('#budgetTableBody');
            tbody.empty();

            const budgets = appData.budgets[key] || [];

            appData.categories.filter(c => c.type === 'expense').forEach(c => {
                const saved  = budgets.find(b => b.categoryId == c.id)?.amount || 0;
                const actual = appData.transactions
                    .filter(t => t.date.startsWith(key) && t.category === c.name && t.type === 'expense')
                    .reduce((s, t) => s + t.amount, 0);

                tbody.append(`
                    <tr>
                        <td><i class="fas fa-circle me-2" style="color:${c.color}"></i>${c.name}</td>
                        <td><input type="number" class="form-control form-control-sm" name="bud_${c.id}" value="${saved}"></td>
                        <td class="text-muted">฿${actual.toLocaleString()}</td>
                        <td class="${saved - actual < 0 ? 'text-danger' : 'text-success'} fw-bold">฿${(saved - actual).toLocaleString()}</td>
                    </tr>`);
            });
        };

        window.copyLastMonthBudget = function () {
            const y = parseInt($('#budgetYearSelect').val());
            const m = parseInt($('#budgetMonthSelect').val());
            let prevY = y, prevM = m - 1;
            if (prevM < 1) { prevM = 12; prevY--; }

            const prevKey    = `${prevY}-${String(prevM).padStart(2, '0')}`;
            const currentKey = `${y}-${String(m).padStart(2, '0')}`;

            if (appData.budgets[prevKey]) {
                appData.budgets[currentKey] = JSON.parse(JSON.stringify(appData.budgets[prevKey]));
                loadBudgetForm();
                Swal.fire('คัดลอกสำเร็จ', `ข้อมูลจากเดือน ${prevM}/${prevY + 543} ถูกนำมาใช้แล้ว`, 'success');
            } else {
                Swal.fire('ไม่พบข้อมูล', 'ไม่มีงบประมาณในเดือนก่อนหน้า', 'info');
            }
        };

        $('#budgetForm').on('submit', function (e) {
            e.preventDefault();
            const key        = `${$('#budgetYearSelect').val()}-${$('#budgetMonthSelect').val()}`;
            const newBudgets = [];

            $('#budgetTableBody tr').each(function () {
                const id  = $(this).find('input').attr('name').split('_')[1];
                const val = parseFloat($(this).find('input').val()) || 0;
                newBudgets.push({ categoryId: id, amount: val });
            });

            appData.budgets[key] = newBudgets;
            saveData();
            Swal.fire('บันทึกเรียบร้อย', '', 'success');
            loadBudgetForm();
        });
    </script>
